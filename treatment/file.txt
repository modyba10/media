import pandas as pd

# Charger les données CSV dans un DataFrame
df = pd.read_csv('data.csv')

# Supprimer les lignes avec les mêmes valeurs dans toutes les colonnes
df = df.drop_duplicates(keep=False)

# Écrire le DataFrame sans doublons dans un nouveau fichier CSV
df.to_csv('data_without_duplicates.csv', index=False)
